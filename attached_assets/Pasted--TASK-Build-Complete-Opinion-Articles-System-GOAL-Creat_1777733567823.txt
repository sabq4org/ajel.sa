// Authors/Columnists table
export const authors = pgTable("authors", {
  id: uuid("id").primaryKey().defaultRandom(),
  fullName: varchar("full_name", { length: 200 }).notNull(),
  slug: varchar("slug", { length: 200 }).notNull().unique(),
  bio: text("bio"),
  shortBio: varchar("short_bio", { length: 300 }), // for cards
  avatarUrl: text("avatar_url"),
  email: varchar("email", { length: 200 }),
  twitter: varchar("twitter", { length: 100 }),
  isActive: boolean("is_active").notNull().default(true),
  position: integer("position").notNull().default(0), // for ordering
  userId: uuid("user_id").references(() => users.id, { onDelete: "set null" }), // optional link to user account
  createdAt: timestamp("created_at").notNull().defaultNow(),
  updatedAt: timestamp("updated_at").notNull().defaultNow(),
});

// Opinion Articles table (separate from regular articles)
export const opinionArticles = pgTable("opinion_articles", {
  id: uuid("id").primaryKey().defaultRandom(),
  slug: varchar("slug", { length: 300 }).notNull().unique(),
  title: varchar("title", { length: 300 }).notNull(),
  subtitle: varchar("subtitle", { length: 500 }),
  excerpt: text("excerpt"),
  contentHtml: text("content_html"),
  contentJson: jsonb("content_json"),
  featuredImageUrl: text("featured_image_url"),
  
  // Author relationship
  authorId: uuid("author_id").notNull().references(() => authors.id),
  
  // Workflow
  status: articleStatusEnum("status").notNull().default("draft"),
  
  // SEO
  metaTitle: varchar("meta_title", { length: 200 }),
  metaDescription: text("meta_description"),
  metaKeywords: text("meta_keywords"),
  
  // Stats
  viewCount: integer("view_count").notNull().default(0),
  commentCount: integer("comment_count").notNull().default(0),
  shareCount: integer("share_count").notNull().default(0),
  readingTimeMinutes: integer("reading_time_minutes"),
  
  // Flags
  isFeatured: boolean("is_featured").notNull().default(false),
  excludeFromHome: boolean("exclude_from_home").notNull().default(false),
  
  // Dates
  publishedAt: timestamp("published_at"),
  scheduledAt: timestamp("scheduled_at"),
  createdAt: timestamp("created_at").notNull().defaultNow(),
  updatedAt: timestamp("updated_at").notNull().defaultNow(),
}, (t) => ({
  authorIdx: index("opinion_author_idx").on(t.authorId),
  statusIdx: index("opinion_status_idx").on(t.status, t.publishedAt),
}));

// Relations
export const authorsRelations = relations(authors, ({ many, one }) => ({
  articles: many(opinionArticles),
  user: one(users, {
    fields: [authors.userId],
    references: [users.id],
  }),
}));

export const opinionArticlesRelations = relations(opinionArticles, ({ one }) => ({
  author: one(authors, {
    fields: [opinionArticles.authorId],
    references: [authors.id],
  }),
}));After schema update:
