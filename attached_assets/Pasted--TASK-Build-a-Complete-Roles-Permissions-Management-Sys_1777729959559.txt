// Roles table
export const roles = pgTable("roles", {
  id: uuid("id").primaryKey().defaultRandom(),
  key: varchar("key", { length: 50 }).notNull().unique(),  // system_admin, supervisor, content_manager, reporter, columnist
  nameAr: varchar("name_ar", { length: 100 }).notNull(),
  nameEn: varchar("name_en", { length: 100 }).notNull(),
  description: text("description"),
  isSystem: boolean("is_system").notNull().default(false), // true for built-in roles (cannot delete)
  level: integer("level").notNull().default(0),            // higher = more powerful (used for hierarchy)
  createdAt: timestamp("created_at").notNull().defaultNow(),
  updatedAt: timestamp("updated_at").notNull().defaultNow(),
});

// Permissions table
export const permissions = pgTable("permissions", {
  id: uuid("id").primaryKey().defaultRandom(),
  key: varchar("key", { length: 100 }).notNull().unique(),  // articles.create, articles.publish, articles.delete...
  nameAr: varchar("name_ar", { length: 200 }).notNull(),
  nameEn: varchar("name_en", { length: 200 }).notNull(),
  category: varchar("category", { length: 50 }).notNull(), // articles, users, media, settings, ai
  description: text("description"),
});

// Role-Permission mapping (many-to-many)
export const rolePermissions = pgTable("role_permissions", {
  roleId: uuid("role_id").notNull().references(() => roles.id, { onDelete: "cascade" }),
  permissionId: uuid("permission_id").notNull().references(() => permissions.id, { onDelete: "cascade" }),
}, (t) => ({
  pk: primaryKey({ columns: [t.roleId, t.permissionId] }),
}));

5️⃣ API ENDPOINTS
═══════════════════════════════════════════════

/api/roles (GET, POST)

• GET: list all roles with user counts and permission counts
• POST: create new role { nameAr, nameEn, description, level } (require roles.create)

/api/roles/[id] (GET, PATCH, DELETE)

• GET: role details + its permissions
• PATCH: update role basic info
• DELETE: delete role (block if isSystem=true OR has assigned users)

/api/roles/[id]/permissions (GET, PUT)

• GET: list of granted permission keys
• PUT: replace entire permissions list { permissions: [...keys] }

/api/permissions (GET)

• GET: list all available permissions grouped by category

/api/users/[id]/role (PATCH)

• PATCH: assign role to user { roleId } (require users.assign_roles)
