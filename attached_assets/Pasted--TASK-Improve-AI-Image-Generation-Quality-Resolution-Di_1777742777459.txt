6️⃣ CLOUDINARY OPTIMIZATION
═══════════════════════════════════════════════

When uploading to Cloudinary, ensure:

// In src/lib/cloudinary.ts upload function:
const uploaded = await cloudinary.uploader.upload(dataUri, {
  folder: 'ai-generated',
  resource_type: 'image',
  format: 'webp',           // ← convert to WebP for performance
  quality: 90,              // ← high quality (was likely auto)
  fetch_format: 'auto',     // ← serve optimal format per browser
  width: 1920,              // ← upscale to consistent width
  crop: 'limit',            // ← maintain aspect, don't enlarge if smaller
  flags: 'progressive',     // ← progressive JPEG/WebP loading
});═══════════════════════════════════════════════
